avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -c voltageOpenTest.c -o voltageOpenTest.o   // compile step
avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -o voltageOpenTest.elf voltageOpenTest.o   // link step 
avr-objcopy -j .text -j .data -O ihex voltageOpenTest.elf voltageOpenTest.hex // generate .hex format file from .elf executable generated by gcc


avrdude -c usbtiny -p t85 -p attiny85 -U flash:w:voltageOpenTest.hex:i // write to the chip
