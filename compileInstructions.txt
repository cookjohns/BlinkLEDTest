avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -c accelTestSmall.c -o accelTestSmall.o   // compile step
avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -o accelTestSmall.elf accelTestSmall.o   // link step 
avr-objcopy -j .text -j .data -O ihex accelTestSmall.elf accelTestSmall.hex // generate .hex format file from .elf executable generated by gcc


avrdude -c usbtiny -p t85 -p attiny85 -U flash:w:accelTestSmall.hex:i // write to the chip
