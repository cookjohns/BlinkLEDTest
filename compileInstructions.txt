avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -c adcTest.c -o adcTest.o   // compile step
avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -o adcTest.elf adcTest.o   // link step

avrdude -c usbtiny -p t85 -p attiny85 -U flash:w:adcTest.hex:i // write to the chip
