avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -c newAdc.c -o newAdc.o   // compile step
avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -o newAdc.elf newAdc.o   // link step 
avr-objcopy -j .text -j .data -O ihex newAdc.elf newAdc.hex // generate .hex format file from .elf executable generated by gcc


avrdude -c usbtiny -p t85 -p attiny85 -U flash:w:newAdc.hex:i // write to the chip
