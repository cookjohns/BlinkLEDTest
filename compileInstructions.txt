avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -c systemMain.c -o systemMain.o   // compile step
avr-gcc -Wall -Os -DF_CPU=1000000 -mmcu=attiny85 -o systemMain.elf systemMain.o   // link step 
avr-objcopy -j .text -j .data -O ihex systemMain.elf systemMain.hex // generate .hex format file from .elf executable generated by gcc


avrdude -c usbtiny -p t85 -p attiny85 -U flash:w:systemMain.hex:i // write to the chip
